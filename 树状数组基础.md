### **一维树状数组**
数组`a[N]`（个数为`n`）与树状区间和节点`tree[N]`
1. **基础函数**
	```cpp
	//获取x二进制下最低位1
	int lowbit(int x){
		return x&-x;
	}
	//在x处加上val：a[x] += val（修改）
	void update(int x,int val){
		while(x<=n){
			tree[x]+=val;
			x+=lowbit(x);
		}
	}
	//前缀和：sum[1..x]（查询）
	int query(int x){
		int sum=0;
		while(x){
			sum+=tree[x];
			x-=lowbit(x);
		}
		return sum;
	}
	```
2. **单点修改，单点查询**
	`tree`维护原数组`a[i]`
	```cpp
	// 单点修改：a[x] += val
	int point_add(int x,int val){
		update(x,val);
	}
	//单点查询:a[x]
	int point_query(int x){
		return query(x)-query(x-1);
	}
	```
3. **单点修改，区间查询**
	`tree`维护原数组`a[i]`
	```cpp
	// 单点修改：a[x] += val
	int point_add(int x,int val){
		update(x,val);
	}
	//区间查询:区间和 [l, r]
	int range_query(int l,int r){
		return query(r)-query(l-1);
	}
	```
4. **区间修改，单点查询**
	`tree`维护差分数组`d[i]`
	```cpp
	//区间修改：区间 [l, r] += val
	void range_add(int l,int r,int val){
		update(l,val); 
		update(r+1,-val);
	}
	//单点查询：a[x]（前缀和->原数组）
	int point_query(int x){
		return query(x);
	}
	```
5. **区间修改，区间查询**
	`tree1`维护差分数组`d[i]`，`tree2`维护`i*d[i]`
	前缀和：$sum[1..x]=(x+1)⋅​∑d_i​−∑(i⋅d_i​)$
	```cpp
	void update(int tree[],int x,int val){
		while(x<=n){
			tree[x]+=val;
			x+=lowbit(x);
		}
	}
	int query(int tree[],int x){
		int sum=0;
		while(x){
			sum+=tree[x];
			x-=lowbit(x);
		}
		return sum;
	}
	// 前缀和 sum[1..x] 
	int pre_sum(int x){
		return (x+1)*query(tree1, x)-query(tree2, x); 
	}
	//区间修改:区间 [l, r] += val
	void range_add(int l,int r,int val){
		update(tree1,l,val);
		update(tree1,r+1,-val);
		update(tree2,l,val*l);
		update(tree2,r+1,-val*(r+1));
	}
	//区间查询:区间和 [l, r]
	int range_query(int l,int r){ 
		return pre_sum(r)-pre_sum(l-1);
	}
	```
### **二维树状数组**
数组`a[N][N]`（个数为`n*n`）与树状区间和节点`tree[N][N]` 
查询结果最保险用`long long`存
1. **基础函数**
	```cpp
	//获取x二进制下最低位1
	int lowbit(int x){
		return x&-x;
	}
	// 单点 (x,y) += val（修改）
	void update(int x,int y,int val){
		for (int i=x;i<n;i+=lowbit(i)){
			for (int j=y;j<N;j+=lowbit(j)){
				tree[i][j] += val;
			}
		}
	}
	// 前缀和 (1,1)~(x,y) （查询）
	int query(int x,int y){
		int sum=0;
			for(int i=x;i>0;i-=lowbit(i)){
				for (int j=y;j>0;j-=lowbit(j)){
					sum+=tree[i][j];
				}
			}
		return sum;
	}
	```
2. **单点修改，单点查询**
	`tree`维护原数组`a[i][j]`
	```cpp
	// 单点修改：a[x][y] += val
	int point_add(int x,int y,int val){
		update(x,y,val);
	}
	//单点查询:a[x][y]
	int point_query(int x,int y){
		return query(x,y)-query(x-1,y)-query(x,y-1)+query(x-1,y-1);
	}
	```
3. **单点修改，矩形查询**
	`tree`维护原数组`a[i][j]`
	```cpp
	// 单点修改：a[x][y] += val
	int point_add(int x,int y,int val){
		update(x,y,val);
	}
	//矩形查询：[x1,y1]~[x2,y2]
	int rect_query(int x1,int y1,int x2,int y2){
		return query(x2,y2)-query(x1-1,y2)-query(x2,y1-1)+query(x1-1,y1-1);
	}
	```
4. **矩形修改，单点查询**
	`tree`维护差分数组`d[i][j]`
	```cpp
	//矩形修改：矩形 [x1,y1]~[x2,y2] + val
	void rect_add(int x1,int y1,int x2,int y2,int val){
		update(x1,y1,val);
		update(x1,y2+1,-val);
		update(x2+1,y1,-val);
		update(x2+1,y2+1,val);
	}
	//单点查询：a[x][y]（前缀和->原数组）
	int point_query(int x,int y){
		return query(x,y);
	}
	```
5. **区间修改，区间查询**
	`tree1`维护差分数组`d[i][j]`，`tree2`维护`i*d[i][j]`
	`tree3`维护差分数组`j*d[i][j]`，`tree4`维护`i*j*d[i][j]`
	前缀和：$sum[(1,1)..(x,y)]=(x+1)⋅(y+1)⋅∑∑d_{i​j}−(y+1)⋅∑∑(i⋅d_{ij}​)-(x+1)⋅∑∑(j⋅d_{ij}​)+∑∑(ij⋅d_{ij}​)$
	```cpp
	void update(int tree[][N],int x,int y,int val){
		for (int i=x;i<n;i+=lowbit(i)){
			for (int j=y;j<N;j+=lowbit(j)){
				tree[i][j] += val;
			}
		}
	}
	int query(int tree[][N],int x,int y){
		int sum=0;
			for(int i=x;i>0;i-=lowbit(i)){
				for (int j=y;j>0;j-=lowbit(j)){
					sum+=tree[i][j];
				}
			}
		return sum;
	}
	// 前缀和 (1,1) ~ (x,y)
	int pre_sum(int x,int y){
		return (x+1)*(y+1)*query(tree1,x,y)-(y+1)*query(tree2,x,y)-(x+1)*query(tree3,x,y)+query(tree4,x,y);
	}
	//矩形修改:矩形 [x1,y1]~[x2,y2] + val
	void rect_add(int x1,int y1,int x2,int y2,int val){
		update(tree1,x1,y1,val);
		update(tree1,x1,y2+1,-val);
		update(tree1,x2+1,y1,-val);
		update(tree1,x2+1,y2+1,val);
		update(tree2,x1,y1,val*x1);
		update(tree2,x1,y2+1,-val*x1);
		update(tree2,x2+1,y1,-val*(x2+1));
		update(tree2,x2+1,y2+1,val*(x2+1));
		update(tree3,x1,y1,val*y1);
		update(tree3,x1,y2+1,-val*(y2+1));
		update(tree3,x2+1,y1,-val*y1);
		update(tree3,x2+1,y2+1,val*(y2+1));
		update(tree4,x1,y1,val*x1*y1);
		update(tree4,x1,y2+1,-val*x1*(y2+1));
		update(tree4,x2+1,y1,-val*(x2+1)*y1);
		update(tree4,x2+1,y2+1,val*(x2+1)*(y2+1));
	}
	//矩形查询: [x1,y1]~[x2,y2]
	int rect_query(int x1,int y1,int x2,int y2){
		return pre_sum(x2,y2)-pre_sum(x1-1,y2)-pre_sum(x2,y1-1)+pre_sum(x1-1,y1-1);
	}
	```


### **用处以及与普通前缀和对比**
1. 适用于对数组的**查询**和**修改**
	- **单点修改 + 区间查询**
	    
	    → **树状数组 完胜** 前缀和 / 差分
	- **只有区间修改 + 区间查询**
	    
	    → 树状数组 ≈ 线段树，**远快于**暴力前缀和
	- **只查询、不修改**
	    
	    → 前缀和最快，树状数组没优势

2. 时间复杂度对比

	*   一维
		- 普通前缀和：   
		    - 预处理：O (n)
		    - **单点修改**：O (n)（要重新算）
		    - 区间查询：O (1)
    
		- 树状数组：
	    - 预处理：O (n)
	    - **单点修改**：O(log n)
	    - 区间查询：O (log n)
    * 二维
		- 普通二维前缀和：
			- 预处理：O (nm)
		    - **单点修改**：O (nm)（全量重算）
		    - 子矩阵查询：O (1)
    
		- 二维树状数组：
		    - 预处理：O (nm)
		    - **单点修改**：O(log n log m)
		    - 子矩阵查询：O (log n log m)
    
3. 什么时候用
	* **只查询，不修改**
	    → 前缀和 / 差分（最简单最快）
	* **单点修改 + 区间 / 子矩阵查询**
	    → 一维 / 二维**树状数组**（代码短、常数小）
	* **区间修改 + 区间查询**
	    → 树状数组（差分思想）或 线段树
	* 数据范围很小（n≤1e3）
	    → 暴力都能过
4. 总结
	- **不修改：前缀和最快**
	- **要修改：树状数组碾压前缀和**
